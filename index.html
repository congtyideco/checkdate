// Khởi tạo Firebase const app = firebase.initializeApp(firebaseConfig); const db = firebase.firestore(); const storage = firebase.storage(); // Lấy các phần tử DOM const webcamElement = document.getElementById('webcam'); const canvasElement = document.getElementById('photo-canvas'); const uploadedImageElement = document.getElementById('uploaded-image'); const cameraPlaceholder = document.getElementById('camera-placeholder'); const startCameraButton = document.getElementById('start-camera-btn'); const takePhotoButton = document.getElementById('take-photo-btn'); const uploadButton = document.getElementById('upload-btn'); const fileInput = document.getElementById('file-input'); const productForm = document.getElementById('product-form'); const productList = document.getElementById('product-list'); const productNameInput = document.getElementById('product-name'); const expiryDateInput = document.getElementById('expiry-date'); const saveButton = document.getElementById('save-btn'); const statusMessage = document.getElementById('status-message'); let stream = null; let photoBlob = null; let photoTaken = false; // Hiển thị thông báo function showMessage(message, type = 'error') { statusMessage.textContent = message; statusMessage.className = 'status-message ' + type; setTimeout(() => { statusMessage.className = 'status-message'; }, 5000); } // Dừng camera function stopCameraStream() { if (stream) { stream.getTracks().forEach(track => track.stop()); stream = null; } } // Xử lý khi bấm nút "Mở Camera" startCameraButton.addEventListener('click', async () => { stopCameraStream(); // Dừng stream cũ nếu có try { // Ưu tiên camera sau trên thiết bị di động stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' } }); webcamElement.srcObject = stream; webcamElement.style.display = 'block'; canvasElement.style.display = 'none'; uploadedImageElement.style.display = 'none'; cameraPlaceholder.style.display = 'none'; startCameraButton.style.display = 'none'; takePhotoButton.style.display = 'inline-block'; uploadButton.style.display = 'none'; saveButton.style.display = 'none'; photoTaken = false; showMessage("Camera đã sẵn sàng. Nhấn 'Chụp Ảnh' để chụp.", 'success'); } catch (err) { console.error("Lỗi khi truy cập camera: ", err); showMessage("Không thể truy cập camera. Vui lòng kiểm tra quyền và thử lại."); // Fallback: sử dụng camera bất kỳ nếu camera sau không khả dụng try { stream = await navigator.mediaDevices.getUserMedia({ video: true }); webcamElement.srcObject = stream; webcamElement.style.display = 'block'; canvasElement.style.display = 'none'; uploadedImageElement.style.display = 'none'; cameraPlaceholder.style.display = 'none'; startCameraButton.style.display = 'none'; takePhotoButton.style.display = 'inline-block'; uploadButton.style.display = 'none'; saveButton.style.display = 'none'; photoTaken = false; showMessage("Camera đã sẵn sàng. Nhấn 'Chụp Ảnh' để chụp.", 'success'); } catch (fallbackErr) { console.error("Lỗi fallback camera: ", fallbackErr); showMessage("Không thể truy cập camera. Vui lòng sử dụng tính năng tải ảnh lên."); } } }); // Xử lý khi bấm nút "Chụp Ảnh" takePhotoButton.addEventListener('click', () => { if (!stream) { showMessage("Camera chưa được khởi động. Vui lòng thử lại."); return; } canvasElement.width = webcamElement.videoWidth; canvasElement.height = webcamElement.videoHeight; const context = canvasElement.getContext('2d'); context.drawImage(webcamElement, 0, 0, canvasElement.width, canvasElement.height); canvasElement.toBlob(blob => { photoBlob = blob; }, 'image/jpeg', 0.95); stopCameraStream(); webcamElement.style.display = 'none'; canvasElement.style.display = 'block'; cameraPlaceholder.style.display = 'none'; takePhotoButton.style.display = 'none'; startCameraButton.style.display = 'inline-block'; uploadButton.style.display = 'inline-block'; saveButton.style.display = 'inline-block'; photoTaken = true; showMessage("Ảnh đã được chụp. Bạn có thể lưu thông tin sản phẩm.", 'success'); }); // Xử lý khi bấm nút "Tải ảnh lên" uploadButton.addEventListener('click', () => { stopCameraStream(); // Dừng camera nếu đang chạy fileInput.click(); }); fileInput.addEventListener('change', (e) => { const file = e.target.files[0]; if (file) { // Kiểm tra xem file có phải là hình ảnh không if (!file.type.match('image.*')) { showMessage("Vui lòng chọn file hình ảnh."); return; } photoBlob = file; const reader = new FileReader(); reader.onload = function(event) { uploadedImageElement.src = event.target.result; uploadedImageElement.style.display = 'block'; webcamElement.style.display = 'none'; canvasElement.style.display = 'none'; cameraPlaceholder.style.display = 'none'; startCameraButton.style.display = 'inline-block'; takePhotoButton.style.display = 'none'; uploadButton.style.display = 'inline-block'; saveButton.style.display = 'inline-block'; photoTaken = true; showMessage("Ảnh đã được tải lên. Bạn có thể lưu thông tin sản phẩm.", 'success'); }; reader.onerror = function() { showMessage("Lỗi khi đọc file. Vui lòng thử lại."); }; reader.readAsDataURL(file); } }); // Xử lý khi bấm nút "Lưu" productForm.addEventListener('submit', async (e) => { e.preventDefault(); if (!photoTaken) { showMessage("Vui lòng chụp ảnh hoặc tải ảnh lên sản phẩm trước khi lưu."); return; } const productName = productNameInput.value.trim(); const expiryDate = expiryDateInput.value; if (!productName) { showMessage("Vui lòng nhập tên sản phẩm."); return; } if (!expiryDate) { showMessage("Vui lòng chọn hạn sử dụng."); return; } saveButton.disabled = true; saveButton.textContent = "Đang lưu..."; const photoName = `images/${Date.now()}_${productName.replace(/[^a-zA-Z0-9]/g, '_')}.jpg`; const storageRef = storage.ref(photoName); try { // Tải ảnh lên Firebase Storage const snapshot = await storageRef.put(photoBlob); const downloadURL = await snapshot.ref.getDownloadURL(); // Lưu thông tin sản phẩm vào Firestore await db.collection("products").add({ name: productName, expiryDate: expiryDate, imageUrl: downloadURL, createdAt: firebase.firestore.FieldValue.serverTimestamp() }); showMessage("Lưu thuốc thành công!", 'success'); productForm.reset(); loadProducts(); // Reset giao diện uploadedImageElement.style.display = 'none'; webcamElement.style.display = 'none'; canvasElement.style.display = 'none'; cameraPlaceholder.style.display = 'block'; startCameraButton.style.display = 'inline-block'; takePhotoButton.style.display = 'none'; uploadButton.style.display = 'inline-block'; saveButton.style.display = 'none'; photoTaken = false; } catch (error) { console.error("Lỗi khi lưu sản phẩm:", error); showMessage("Lỗi khi lưu sản phẩm. Vui lòng thử lại."); } finally { saveButton.disabled = false; saveButton.textContent = "Lưu"; } }); // Tải và hiển thị danh sách thuốc async function loadProducts() { try { const snapshot = await db.collection("products") .orderBy("createdAt", "desc") .get(); productList.innerHTML = ''; if (snapshot.empty) { productList.innerHTML = '

Chưa có thuốc nào được lưu.

'; return; } snapshot.forEach(doc => { const product = doc.data(); const productElement = document.createElement('div'); productElement.className = 'product-item'; productElement.innerHTML = `
${product.name}

HSD: ${new Date(product.expiryDate).toLocaleDateString('vi-VN')}

`; productList.appendChild(productElement); }); } catch (error) { console.error("Lỗi khi tải danh sách sản phẩm:", error); productList.innerHTML = '

Lỗi khi tải danh sách sản phẩm.

'; } } // Tải danh sách sản phẩm khi trang được tải document.addEventListener('DOMContentLoaded', loadProducts);
